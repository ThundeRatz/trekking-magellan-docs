62 /**
63  * Initialize UART.
64  * Initialize UART0 for receiving commands. Uses pins PA0 and PA1 and 2400 baud.
65  * Its interrupt has priority 1.
66  */
67 void UART_init() {
68  // Configure PA0, PA1 as UART
69  ROM_GPIOPinConfigure(GPIO_PA0_U0RX);
70  ROM_GPIOPinConfigure(GPIO_PA1_U0TX);
71  ROM_GPIOPinTypeUART(GPIO_PORTA_BASE, GPIO_PIN_0 | GPIO_PIN_1);
72 
73  ROM_UARTConfigSetExpClk(UART0_BASE, SYSTEM_CLOCK, 2400,
74  (UART_CONFIG_WLEN_8 | UART_CONFIG_STOP_ONE | UART_CONFIG_PAR_NONE));
75 
76  ROM_IntPrioritySet(INT_UART0, 1 << 5);
77  ROM_IntEnable(INT_UART0);
78  ROM_UARTIntEnable(UART0_BASE, UART_INT_OE | UART_INT_BE |
79  UART_INT_PE | UART_INT_FE | UART_INT_RT | UART_INT_RX);
80 }

102 /**
103  * UART Interrupt Service Routine.
104  * Calls motors() with received commands.
105  * @see motors
106  */
107 void UART_ISR() {
108  uint32_t int_status;
109  leds_b(4000);
110  int_status = ROM_UARTIntStatus(UART0_BASE, true);
111  ROM_UARTIntClear(UART0_BASE, int_status);
112 
113  // Iterate over each status bit
114  for (; int_status; int_status &= int_status - 1) {
115  switch (int_status & -int_status) {
116  case UART_INT_RX:
117  watchdog_reset();
118  while (ROM_UARTCharsAvail(UART0_BASE))
119  motors(ROM_UARTCharGet(UART0_BASE));
120  break;
121 
122  case UART_INT_OE: case UART_INT_BE:
123  case UART_INT_PE: case UART_INT_FE:
124  case UART_INT_RT:
125  leds_status(4095);
126 #ifdef DEBUG
127  error_count[__builtin_clz(int_status)]++;
128 #endif
129  break;
130  }
131  }
132 }
