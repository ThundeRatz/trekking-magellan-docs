64 /**
65  * Initialize motors PWM output.
66  * Configure M0PWM2, M0PWM3, M0PWM7 and M0PWM8 at PB4, PB5, PC4 and PC5,
67  * respectively. Notice that PWMs are limited to 64/max_duty_cicle operation
68  * (in our case, 2/3 or 1/2, because we use 16V or 12V motors with a 24V battery)
69  * The max. duty cicle can be configured at the source file (see
70  * ROM_PWMGenPeriodSet at motors.c). The motor controller board has pull-down
71  * resistors at FET drivers input to drive outputs low while the Tiva is off/
72  * initializing.
73  */
74 void motors_init() {
75  // Configure PB4, PB5, PC4, PC5 as PWM
76  ROM_GPIOPinConfigure(GPIO_PB4_M0PWM2);
77  ROM_GPIOPinConfigure(GPIO_PB5_M0PWM3);
78  ROM_GPIOPinConfigure(GPIO_PC4_M0PWM6);
79  ROM_GPIOPinConfigure(GPIO_PC5_M0PWM7);
80  ROM_GPIOPinTypePWM(GPIO_PORTB_BASE, GPIO_PIN_4 | GPIO_PIN_5);
81  ROM_GPIOPinTypePWM(GPIO_PORTC_BASE, GPIO_PIN_4 | GPIO_PIN_5);
82 
83  // PWM_GEN_1 covers M0PWM2 and M0PWM3
84  // PWM_GEN_3 covers M0PWM6 and M0PWM7
85  ROM_PWMGenConfigure(PWM0_BASE, PWM_GEN_1, PWM_GEN_MODE_UP_DOWN | PWM_GEN_MODE_NO_SYNC |
86  PWM_GEN_MODE_DB_NO_SYNC| PWM_GEN_MODE_FAULT_UNLATCHED | PWM_GEN_MODE_FAULT_NO_MINPER);
87  ROM_PWMGenConfigure(PWM0_BASE, PWM_GEN_3, PWM_GEN_MODE_UP_DOWN | PWM_GEN_MODE_NO_SYNC |
88  PWM_GEN_MODE_DB_NO_SYNC | PWM_GEN_MODE_FAULT_UNLATCHED | PWM_GEN_MODE_FAULT_NO_MINPER);
89 
90  // Set the Period (in clock ticks)
91  // Frequency = (SYSTEM_CLOCK / PWM_CLOCK_DIV) / (ui32Gen)
92  // Period (seconds) = PWM_CLOCK_DIV * ui32Gen / SYSTEM_CLOCK
93  // Here, frequency = 1953.125
94  ROM_PWMGenPeriodSet(PWM0_BASE, PWM_GEN_1, 128);
95  ROM_PWMGenPeriodSet(PWM0_BASE, PWM_GEN_3, 128);
96 
97  // Enable the PWM generator
98  ROM_PWMGenEnable(PWM0_BASE, PWM_GEN_1);
99  ROM_PWMGenEnable(PWM0_BASE, PWM_GEN_3);
100 
101  // Enable PWM
102  ROM_PWMOutputState(PWM0_BASE, PWM_OUT_2_BIT | PWM_OUT_3_BIT | PWM_OUT_6_BIT | PWM_OUT_7_BIT, true);
103 }

105 /**
106  * Parse UART0 command.
107  * Parse command received from UART0 and set speed_left and speed_right
108  * accordingly.
109  * @param[in] command
110  * @parblock
111  * command received. Bit 7 chooses the motor (0 = left, 1 = right). Bit 6
112  * reverses direction. Bits 0..5 choose speed.
113  * @endparblock
114  */
115 void motors(uint8_t command) {
116  if (command & MOTOR_RIGHT)
117  if (command & MOTOR_REV)
118  speed_right = -(command & 0x3f);
119  else
120  speed_right = command & 0x3f;
121  else
122  if (command & MOTOR_REV)
123  speed_left = -(command & 0x3f);
124  else
125  speed_left = command;
126 }

128 /**
129  * Set left motor PWM.
130  * Set PWM duty cicle.
131  * @param[in] pwm_value new value
132  */
133 void motor_left(int8_t pwm_value) {
134  if (pwm_value < 0) {
135  ROM_PWMPulseWidthSet(PWM0_BASE, PWM_OUT_2, -pwm_value);
136  ROM_PWMPulseWidthSet(PWM0_BASE, PWM_OUT_3, 0);
137  } else {
138  ROM_PWMPulseWidthSet(PWM0_BASE, PWM_OUT_2, 0);
139  ROM_PWMPulseWidthSet(PWM0_BASE, PWM_OUT_3, pwm_value);
140  }
141 }
